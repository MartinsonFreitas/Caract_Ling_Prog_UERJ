public class Pessoa implements Cloneable {
	private String nome;

public Pessoa(String nome) {
	this.nome = nome;
}

public String getNome() {
	return nome;
}

public void setNome(String nome) {
	this.nome = nome;
}

// código do método clone incluído pelo prof. Oscar
@Override
public Pessoa clone() throws CloneNotSupportedException { 
        Pessoa p;

	p = (Pessoa) super.clone();
	p.nome = new String(nome);   
	return p;
} 
}
